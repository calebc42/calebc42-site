Configuration

(use-package helpful
  :ensure t
  :demand t
  :bind
  ;; Remap the default help keys to their helpful equivalents
  (("C-h f" . helpful-function)
   ("C-h v" . helpful-variable)
   ("C-h k" . helpful-key)
   ("C-h c" . helpful-command)
   ("C-h m" . helpful-macro)  
   ("C-c h" . helpful-at-point)))
