;; --- Gatekeeper Functions ---
(defun my/org-publish-if-publish-property (_plist filename _pub-dir)
  "Publish FILENAME if it has the property :publish: set to `t`, or if it's the sitemap."
  (let ((basename (file-name-nondirectory filename)))
    (or (string-equal basename "sitemap.org")  ; Always publish sitemap
        (with-temp-buffer
          (insert-file-contents filename)
          (goto-char (point-min))
          (org-mode)
          (let ((publish-prop (org-entry-get (point-min) "publish")))
            (and publish-prop (string-equal publish-prop "t")))))))

(defun my/org-publish-if-tutorial-property (_plist filename _pub-dir)
  "Publish FILENAME if it has the property :type: set to `tutorial`."
  (with-temp-buffer
    (insert-file-contents filename)
    (goto-char (point-min))
    (org-mode)
    (let ((type-prop (org-entry-get (point-min) "type")))
      (and type-prop (string-equal type-prop "tutorial")))))

;; --- Unified Postamble Function ---
(defun my/site-html-postamble (path-prefix plist)
  "Generate a dynamic postamble with correct relative paths for scripts.
PATH-PREFIX is the relative path to the assets directory, e.g., 'assets/' or '../assets/'."
  (let* ((input-file (plist-get plist :input-file))
         (mod-time (nth 5 (file-attributes input-file)))
         (date-str (format-time-string "%Y-%m-%d" mod-time)))
    (concat
     "<footer class=\"main-footer\">"
     (format "<p>Last Modified: %s</p>" date-str)
     "<p>This site is built with Emacs and Org-mode.</p>"
     "<p>// SYSTEM_NORMAL // [C] 2025 CALEB CHRISTENSEN</p>"
     "</footer>"
     ;; Add scripts with the correct relative path
     (format "<script src=\"%stheme.js\"></script>" path-prefix)
     (format "<script src=\"https://platform.linkedin.com/badges/js/profile.js\" async defer type=\"text/javascript\"></script>"))))

;; NOTE: These are only active if you use blocks like #+begin_yorha-panel

;; NOTE: Prefixed unused arguments with '_' to silence compiler warnings.
;; The Org export framework requires these arguments in the function
;; signature, even if they aren't all used in the function body.

(defun org-html-yorha-panel (panel contents _info)
  "Export PANEL block to HTML with YoRHa styling."
  (let ((title (or (plist-get (org-export-read-attribute :attr_html panel) :title) "DATA")))
    (format "<div class=\"yorha-panel\" data-title=\"%s\">%s</div>" title contents)))

(defun org-html-yorha-terminal (_terminal contents _info)
  "Export TERMINAL block to HTML with terminal styling."
  (format "<div class=\"yorha-terminal\">%s</div>" contents))

(defun org-html-yorha-grid (_grid contents _info)
  "Export GRID block to HTML with grid layout."
  (format "<div class=\"yorha-grid\">%s</div>" contents))

(defun org-html-yorha-grid-item (_item contents _info)
  "Export GRID-ITEM block to HTML."
  (format "<div class=\"yorha-grid-item\">%s</div>" contents))
