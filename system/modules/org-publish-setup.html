Configuration

This section defines the entire publishing workflow, including the gatekeeper function for selective publishing and the project alist that specifies source and destination paths. 

  ;; This makes all publishing functions and variables known to the compiler.
  (require 'ox-publish)

  ;; --- 1. Define the "gatekeeper" function for selective publishing ---
(defun my/org-publish-if-publish-property (_plist filename _pub-dir)
  "Publish FILENAME if it has the property :publish: set to `t`, or if it's the sitemap."
  (let ((basename (file-name-nondirectory filename)))
    (or (string-equal basename "sitemap.org")  ; Always publish sitemap
        (with-temp-buffer
          (insert-file-contents filename)
          (goto-char (point-min))
          (org-mode)
          (let ((publish-prop (org-entry-get (point-min) "publish")))
            (and publish-prop (string-equal publish-prop "t")))))))

  ;; --- 2. Define the publishing project ---
(setq org-publish-project-alist
  '(("coppermind-org" 
     :base-directory "~/coppermind/"
     :publishing-directory "~/site/" 
     :base-extension "org" 
     :recursive t
     :publishing-function org-html-publish-to-html 
     :publish-p my/org-publish-if-publish-property   
     :headline-levels 4
     :auto-sitemap t 
     :sitemap-filename "sitemap.org" 
     :sitemap-title "Sitemap"
     :html-head-include-default-style nil 
     :html-head-include-scripts nil
     :html-head "<link rel=\"stylesheet\" type=\"text/css\" href=\"assets/style.css\"/>"
     :section-numbers nil 
     :with-toc t 
     :toc-depth 2)
    ("coppermind-assets" 
     :base-directory "~/coppermind/resources/site/assets/"
     :publishing-directory "~/site/assets/" 
     :base-extension "css\\|js\\|png\\|jpg\\|gif\\|svg"
     :recursive t 
     :publishing-function org-publish-attachment)  
    ("coppermind" 
     :components ("coppermind-org" "coppermind-assets"))))
