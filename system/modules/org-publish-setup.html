Configuration

This section defines the entire publishing workflow, including the gatekeeper function for selective publishing and the project alist that specifies source and destination paths. 

;; This makes all publishing functions and variables known to the compiler.
(require 'ox-publish)

;; --- 1. Define the "gatekeeper" function for selective publishing ---
(defun my/org-publish-if-publish-property (_plist filename _pub-dir)
  "Publish FILENAME if it has the property :publish: set to `t`, or if it's the sitemap."
  (let ((basename (file-name-nondirectory filename)))
    (or (string-equal basename "sitemap.org")  ; Always publish sitemap
        (with-temp-buffer
          (insert-file-contents filename)
          (goto-char (point-min))
          (org-mode)
          (let ((publish-prop (org-entry-get (point-min) "publish")))
            (and publish-prop (string-equal publish-prop "t")))))))

YoHRa UI Custom Blocks

;; Custom export blocks for YoRHa elements
(defun org-html-yorha-panel (panel contents info)
  "Export PANEL block to HTML with YoRHa styling."
  (let ((title (or (plist-get (org-export-read-attribute :attr_html panel) :title) "DATA")))
    (format "<div class=\"yorha-panel\" data-title=\"%s\">%s</div>" title contents)))

(defun org-html-yorha-terminal (terminal contents info)
  "Export TERMINAL block to HTML with terminal styling."
  (format "<div class=\"yorha-terminal\">%s</div>" contents))

(defun org-html-yorha-grid (grid contents info)
  "Export GRID block to HTML with grid layout."
  (format "<div class=\"yorha-grid\">%s</div>" contents))

(defun org-html-yorha-grid-item (item contents info)
  "Export GRID-ITEM block to HTML."
  (format "<div class=\"yorha-grid-item\">%s</div>" contents))

;; Register the custom blocks
(org-export-define-derived-backend 'yorha-html 'html
  :translate-alist '((special-block . org-html-special-block-yorha)))

(defun org-html-special-block-yorha (special-block contents info)
  "Handle custom YoRHa blocks."
  (let ((block-type (org-element-property :type special-block)))
    (cond
     ((string= block-type "yorha-panel") (org-html-yorha-panel special-block contents info))
     ((string= block-type "yorha-terminal") (org-html-yorha-terminal special-block contents info))
     ((string= block-type "yorha-grid") (org-html-yorha-grid special-block contents info))
     ((string= block-type "yorha-grid-item") (org-html-yorha-grid-item special-block contents info))
     (t (org-html-special-block special-block contents info)))))

(defun org-html-data-panel (panel contents info)
  "Export data panel block with your aesthetic."
  (let ((title (or (plist-get (org-export-read-attribute :attr_html panel) :title) "DATA")))
    (format "<div class=\"data-panel\" data-title=\"%s\">%s</div>" title contents)))

(defun org-html-content-grid (grid contents info)
  "Export content grid using your spacing system."
  (format "<div class=\"content-grid\">%s</div>" contents))

(defun org-html-grid-item (item contents info)
  "Export grid item with your styling."
  (format "<div class=\"grid-item\">%s</div>" contents))
