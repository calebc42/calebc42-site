Configuration

(use-package corfu
  :ensure t
  ;; Trigger completion automatically as you type.
  :init
  (global-corfu-mode)
  ;; Start completion after a short delay.
  (setq corfu-auto-delay 0.2)
  ;; Customize the behavior.
  :custom
  (corfu-cycle t)                ; Allow cycling through candidates with TAB.
  (corfu-auto t)                 ; Enable auto-completion.
  (corfu-separator ?\s)          ; Use a space to separate preview from input.
  (corfu-quit-at-boundary 'never); Let completion continue across word boundaries.
  (corfu-preview-current 'insert) ; Show preview of current candidate
  (corfu-preselect 'prompt)       ; Don't preselect first candidate
  :bind
  (:map corfu-map
        ("TAB" . corfu-complete)
          ("RET" . corfu-insert)))

;;; corfu-setup.el ends here
