2. Configuration

(use-package org-roam
  :ensure t
  :init
  ;; Ensure emacsql can find the sqlite3 executable.
  (when (executable-find "sqlite3")
    (setq emacsql-sqlite-executable (executable-find "sqlite3")))
  :after org
  :hook (after-init . org-roam-db-autosync-mode)
  :custom
  (org-roam-directory (file-truename "~/coppermind"))
  (org-roam-completion-everywhere t)
  (org-roam-node-display-template
   (concat "${title:*} " (propertize "${tags:10}" 'face 'org-tag)))
  :bind (("C-c n f" . org-roam-node-find)
         ("C-c n i" . org-roam-node-insert)
         ("C-c n c" . org-roam-capture)
         ("C-c n b" . org-roam-buffer-toggle)
         ("C-c n p" . coppermind/set-property-interactively))
  :config
  ;; Keep the database in a central, known location.
  (setq org-roam-db-location (expand-file-name "org-roam.db" user-emacs-directory))
  ;; --- DYNAMIC CAPTURE TEMPLATES ---
  ;; These templates call functions from coppermind-lib.el to generate
  ;; files with a property drawer based on the schema in ~/coppermind/system/properties/
  (setq org-roam-capture-templates
        '(("r" "resource" plain
           (function coppermind/resource-template)  ; <- Fixed: no arguments
           :target (file "resources/%<%Y%m%d%H%M%S>-${slug}.org")
           :unnarrowed t)
          ("p" "project" plain
           (function coppermind/project-template)   ; <- Fixed: no arguments
           :target (file "projects/%<%Y%m%d%H%M%S>-${slug}.org")
           :unnarrowed t)
          ("i" "inbox" plain "* %?"
           :target (file+olp "inbox.org" ("Inbox"))
           :unnarrowed t)
          ("c" "Company" plain "* %^{Company Name}"
           :target (file+head "career/companies/${slug}.org"
        "#+title: ${title}\n#+filetags: :company:\n\n:PROPERTIES:\n:WEBSITE: %^{Website}\n:INDUSTRY: %^{Industry}\n:END:\n\n** About\n\n%?\n\n** Benefits\n\n** Address\n")
           :unnarrowed t)
               ("j" "Job Application" plain
             "* TODO %^{Role} at %(calebc42/org-roam-find-company-link)"
           :target (file+head (format "career/applications/%s-%s.org"
                                      (format-time-string "%Y%m%d")
                                      "${slug}")
                              "#+title: ${title}\n#+filetags: :application:\n\n:PROPERTIES:\n:URL: %^{URL}\n:SALARY: %^{Salary}\n:LOCATION: %^{Location}\n:EXCITEMENT: %^{Excitement (1-5)}\n:APPLIED_ON:\n:END:\n\n** Responsibilities\n%?\n\n** Qualifications\n\n** Tasks\n- [ ] Tailor Resume\n- [ ] Write Cover Letter\n- [ ] Submit Application\n- [ ] Follow Up\n\n** Notes\n")
           :unnarrowed t)))

  )
