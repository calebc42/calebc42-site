Dashboard Generation

(defun coppermind/create-dashboard-file ()
  "Create or overwrite the main Dashboard.org file in the coppermind root."
  (interactive)
  (let ((dashboard-file (expand-file-name "~/coppermind/Dashboard.org")))
    (with-temp-buffer
      (insert "#+title: Coppermind Dashboard\n\n")
      (insert "* Processing Queue\n\n")
      (insert "** Inbox Items\n")
      (insert "#+begin: clocktable :maxlevel 2 :scope file:\"~/coppermind/inbox.org\"\n#+end:\n\n")
      (insert "** Reviews Due\n")
      (insert "#+begin: org-roam-ql\n")
      (insert "  (and (todo \"TODO\") (pred<(org-roam-node-prop ?n \"REVIEW_DUE\") (org-read-date nil t \"<today>\")))\n")
      (insert "#+end:\n\n")

      (insert "* Projects\n\n")
      (insert "** Next Actions\n")
      (insert "#+begin: org-roam-ql\n")
      (insert "  (and (todo \"TODO\") (prop \"STATUS\" \"next\"))\n")
      (insert "#+end:\n\n")
      (insert "** Active Projects\n")
      (insert "#+begin: org-roam-ql\n")
      (insert "  (and (todo \"TODO\") (prop \"STATUS\" \"active\"))\n")
      (insert "#+end:\n\n")

      (insert "* Resources\n\n")
      (insert "** Recently Modified\n")
      (insert "#+begin: org-roam-ql\n")
      (insert "  (and (prop \"TYPE\" \"resource\") (sort-by 'mtime >))\n")
      (insert "#+end:\n\n")

      ;; Write the buffer content to the file
      (write-file dashboard-file))
    (message "Coppermind Dashboard created/updated.")))

;; Create the dashboard when this module is loaded.
(coppermind/create-dashboard-file)
