2. Helper Functions

These functions streamline the process of creating and finding configuration modules. 

(defun my/new-config-module (package-name)
  "Create a new configuration module for PACKAGE-NAME.
Prompts for a location based on the centrally defined
`my/config-subdirectories'."
  (interactive "sPackage or Library name: ")
  (let* ((dir-choice
          (completing-read "Create in: " my/config-subdirectories nil t))
         (destination-dir
          (expand-file-name (concat dir-choice "/") my/org-config-root-path))
         ;; Add a standard suffix for consistency
         (filename (concat package-name "-setup.org")))
    (find-file (expand-file-name filename destination-dir))))

(defun my/goto-config-directory ()
  "Quickly navigate to one of the config subdirectories."
  (interactive)
  (let* ((dir-choice (completing-read "Go to: " my/config-subdirectories nil t))
         (destination-dir (expand-file-name (concat dir-choice "/") my/org-config-root-path)))
    (dired destination-dir)))

;; Keybindings
(global-set-key (kbd "C-c m n") #'my/new-config-module)
(global-set-key (kbd "C-c m d") #'my/goto-config-directory)

3. Template Functions

(defun my/insert-config-module-template ()
  "Insert boilerplate for a literate config module."
  (let* ((full-path (buffer-file-name))
         (filename (file-name-nondirectory full-path))
         (module-name (file-name-sans-extension filename))
         ;; Remove -setup suffix if present to get package name
         (package-name (if (string-suffix-p "-setup" module-name)
                          (substring module-name 0 -6)
                        module-name))
         ;; Determine which subdirectory we're in
         (sub-dir (cl-find-if
                  (lambda (dir) 
                    (string-match-p (concat "/" dir "/") full-path))
                  my/config-subdirectories))
         (tangle-path (format "~/.emacs.d/lisp/%s/%s.el" sub-dir module-name))
         (display-name (capitalize (replace-regexp-in-string "-" " " package-name)))
         ;; Add these for the properties drawer
         (id (org-id-uuid))
         (date (format-time-string "[%Y-%m-%d %a %H:%M]")))

    ;; Use concat instead of format to avoid issues with embedded syntax
    (insert 
     (concat
      ":PROPERTIES:\n"
      ":ID:       " id "\n"
      ":type:     config\n"
      ":tags:\n"
      ":archived: f\n"
      ":modified: " date "\n"
      ":END:\n\n"
      "#+TITLE: " display-name " Configuration\n"
      "#+DESCRIPTION: Configuration for the " package-name " package.\n"
      "#+PROPERTY: PROVIDES " module-name "\n"
      "#+PROPERTY: LOAD_AFTER \n"
      "#+PROPERTY: header-args :tangle " tangle-path "\n"
      "#+FILETAGS: \n\n"
      "* Header\n"
      "#+begin_src emacs-lisp\n"
      ";;; " module-name ".el --- Configuration for " display-name " -*- lexical-binding: t -*-\n\n"
      ";;; Commentary:\n"
      ";; This file is tangled from ~/coppermind/system/" sub-dir "/" filename ".\n"
      ";; Configuration for the " package-name " package.\n\n"
      ";;; Code:\n"
      "#+end_src\n\n"
      "* Configuration\n"
      "#+begin_src emacs-lisp\n"
      "(use-package " package-name "\n"
      "  :ensure t\n"
      "  :config\n"
      "  ;; Add your configuration here\n"
      "  )\n"
      "#+end_src\n\n"
      "* Footer\n"
      "#+begin_src emacs-lisp\n"
      "(provide '" module-name ")\n"
      ";;; " module-name ".el ends here\n"
      "#+end_src\n"))))

(defun my/insert-default-org-template ()
  "Insert default Org-mode template with dynamic values."
  (let* ((file-path (buffer-file-name))
         (file-name (file-name-nondirectory file-path))
         (base-name (file-name-sans-extension file-name))
         (title (capitalize (replace-regexp-in-string "-" " " base-name)))
         (id (org-id-uuid))
         (date (format-time-string "[%Y-%m-%d %a %H:%M]")))
    (insert 
     (concat
      ":PROPERTIES:\n"
      ":ID:       " id "\n"
      ":type:     \n"
      ":tags:\n"
      ":archived: f\n"
      ":modified: " date "\n"
      ":END:\n\n"
      "#+TITLE: " title "\n"
      "#+AUTHOR: Caleb Christensen\n"
      "#+DESCRIPTION: \n"
      "#+FILETAGS:\n\n\n"))))

4. Auto-insert Configuration

(use-package autoinsert
  :ensure nil ; Built-in package
  :demand t
  :init
  ;; Set the templates directory
  (setq auto-insert-directory (expand-file-name "templates/" "~/coppermind/resources/"))

  :config
  ;; Clear existing auto-insert rules to avoid conflicts
  (setq auto-insert-alist nil)

  ;; Rule 1: Config modules in system directories
  (let ((config-dirs-pattern 
         (concat (regexp-quote my/org-config-root-path)
                 "\\(" (string-join my/config-subdirectories "\\|") "\\)/"
                 ".*\\.org\\'")))
    (add-to-list 'auto-insert-alist
                 `(,config-dirs-pattern . my/insert-config-module-template)))

  ;; Rule 2: Default template for all other .org files
  (add-to-list 'auto-insert-alist
               '("\\.org\\'" . my/insert-default-org-template))

  ;; Enable auto-insert mode
  (auto-insert-mode 1)

  ;; Don't prompt before insertion
  (setq auto-insert-query nil))
