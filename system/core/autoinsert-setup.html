Helper Functions

These functions streamline the process of creating and finding configuration modules. 

(defun my/new-config-module (package-name)
  "Create a new configuration module for PACKAGE-NAME.
Prompts for a location based on the centrally defined
`my/config-subdirectories'."
  (interactive "sPackage or Library name: ")
  (let* ((dir-choice
          ;; Use the variable from config.el as the source for completion options
          (completing-read "Create in: " my/config-subdirectories nil t))
         (destination-dir
          (expand-file-name (concat dir-choice "/") my/org-config-root-path))
         ;; Add a standard suffix for consistency
         (filename (concat package-name "-setup.org")))
    (find-file (expand-file-name filename destination-dir))))

;; Bind this to a convenient key for quick access
(global-set-key (kbd "C-c m n") #'my/new-config-module)

(defun my/goto-config-directory ()
  "Quickly navigate to one of the config subdirectories."
  (interactive)
  (let* ((dir-choice (completing-read "Go to: " my/config-subdirectories nil t))
         (destination-dir (expand-file-name (concat dir-choice "/") my/org-config-root-path)))
    (dired destination-dir)))

(global-set-key (kbd "C-c m d") #'my/goto-config-directory)

Auto-insert Configuration

This section configures the `auto-insert` package to use our dynamic template. 

(use-package autoinsert
  :ensure nil ; It's a built-in package
  :demand t
  :config
  (defun my/insert-config-module-header ()
    "Insert boilerplate for a literate config by reading a template file
and formatting it with dynamic values."
    (interactive)
    (let* ((full-path (buffer-file-name))
           (filename (file-name-nondirectory full-path))
           (module-name (file-name-sans-extension filename))
           (package-name (if (string-suffix-p "-setup" module-name)
                             (substring module-name 0 -6)
                           module-name))
           (sub-dir (car (cl-remove-if-not
                         (lambda (dir) (string-match-p (concat "/" dir "/") full-path))
                         my/config-subdirectories)))
           (tangle-path (format "~/.emacs.d/lisp/%s/%s.el" sub-dir module-name))
           (display-name (capitalize (replace-regexp-in-string "-" " " package-name)))
           (template-path (expand-file-name "config-module.org" "~/coppermind/resources/templates/"))
           (template-string (with-temp-buffer
                              (insert-file-contents-literally template-path)
                              (buffer-string))))
      (insert
       (format
        template-string
        ;; Arguments to format string (these are the same as before):
        display-name      ; Title
        package-name      ; Description
        package-name      ; PROVIDES
        tangle-path       ; Tangle path
        module-name       ; File header
        display-name      ; File header comment
        sub-dir           ; Dynamic path part
        filename          ; Commentary path
        package-name      ; use-package name
        module-name))))   ; Footer comment

  ;; --- Tell Emacs where to find template files in general ---
  (setq auto-insert-directory (expand-file-name "templates/" "~/coppermind/system/"))

  ;; --- RULE 1: For your special config files, use the dynamic function ---
  (let* ((config-path-regex (regexp-quote my/org-config-root-path))
         (subdirs-regex (string-join my/config-subdirectories "\\|")))
    (add-to-list 'auto-insert-alist
                 `(,(format "%s\\(%s\\)/.+\\.org\\'" config-path-regex subdirs-regex)
                   . my/insert-config-module-header)))

  ;; --- RULE 2: For any other .org file, use the simple default template ---
  (add-to-list 'auto-insert-alist '("\\.org\\'" . "default.org"))

  (auto-insert-mode 1))
