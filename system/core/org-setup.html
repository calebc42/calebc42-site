Org Mode Core

This `use-package` block configures the core behavior of Org Mode. It sets default directories, enables Babel languages, and defines some quality-of-life settings. 

(use-package org
      :ensure t
      :pin gnu
      :demand t
      ;; We don't need to :ensure org because it's built into modern Emacs.
      :config
      ;; --- 1. Babel Languages ---
      ;; This is the crucial part to ensure that Emacs Lisp source blocks
      ;; are executable and can be tangled correctly.
      (org-babel-do-load-languages
       'org-babel-load-languages
       '((emacs-lisp . t)
         (shell . t))) ; It's useful to enable shell scripts as well.

      ;; --- 2. Core Settings ---
      (setq org-directory             "~/coppermind"
            org-agenda-files          '("~/coppermind")
            org-default-notes-file    (expand-file-name "inbox.org" org-directory)
            org-ellipsis              " ▾" ; A nicer folding indicator.
            org-startup-indented      t    ; Prettier indentation at startup.
            org-log-done              'time ; Record when a TODO item is marked DONE.
    ))

;;; org-setup.el ends here
